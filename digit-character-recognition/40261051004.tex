	\section{مقدمه}
	در این گزارش، نتایج مربوط به تمرین دوم درس یادگیری عمیق شامل بررسی تاثیر حذف لایه
	Pooling و جایگزینی آن با تنظیمات Stride در شبکه‌های عصبی پیچشی (CNN) بررسی شده
	است.
	
	\section{بخش اول: بررسی داده‌های EMNIST}
	همان‌طور که در صورت سوال خواسته شده بود، برای بررسی تاثیر تنظیمات مختلف، ابتدا
	یک مدل پایه شامل لایه \lr{Max Pooling} طراحی گردید. سپس با حذف این لایه و تغییر
	گام حرکت (\lr{Stride}) و پدینگ (\lr{Padding}) در لایه‌های کانولوشن، عملکرد مدل
	ارزیابی شد.
	
	\subsection{کد پیاده‌سازی مدل}
	در قطعه کد زیر، فرآیند کامل شامل بارگذاری داده‌ها، تابع ساخت مدل‌های مختلف و حلقه
	اجرای تست‌ها آورده شده است:
	
	\begin{latin}
		\begin{lstlisting}
import tensorflow as tf
from tensorflow.keras import layers, models
import tensorflow_datasets as tfds
import time

def load_emnist():
print("Loading EMNIST dataset...")
(ds_train, ds_test), ds_info = tfds.load(
'emnist/balanced',
split=['train', 'test'],
shuffle_files=True,
as_supervised=True,
with_info=True,
)

def normalize_img(image, label):
image = tf.cast(image, tf.float32) / 255.0
image = tf.transpose(image, [1, 0, 2])
return image, label

batch_size = 128
ds_train = ds_train.map(normalize_img).cache().shuffle(10000).batch(batch_size).prefetch(tf.data.AUTOTUNE)
ds_test = ds_test.map(normalize_img).batch(batch_size).cache().prefetch(tf.data.AUTOTUNE)

return ds_train, ds_test

def build_model(use_pooling=True, conv_stride=(1, 1), padding_type='valid'):
model = models.Sequential()

model.add(layers.InputLayer(shape=(28, 28, 1)))
model.add(layers.Conv2D(32, (3, 3), activation='relu', padding=padding_type))

if use_pooling:
model.add(layers.MaxPooling2D((2, 2)))
else:
model.add(layers.Conv2D(32, (3, 3), strides=conv_stride, activation='relu', padding=padding_type))

model.add(layers.Conv2D(64, (3, 3), activation='relu', padding=padding_type))

if use_pooling:
model.add(layers.MaxPooling2D((2, 2)))
else:
model.add(layers.Conv2D(64, (3, 3), strides=conv_stride, activation='relu', padding=padding_type))

model.add(layers.Flatten())
model.add(layers.Dense(128, activation='relu'))
model.add(layers.Dropout(0.3))
model.add(layers.Dense(47, activation='softmax'))

model.compile(optimizer='adam',
loss='sparse_categorical_crossentropy',
metrics=['accuracy'])
return model
		\end{lstlisting}
	\end{latin}

	با بررسی داده‌های جدول، می‌توان به نتایج زیر رسید:
	
	\begin{itemize}
		\item \textbf{بررسی دقت (Accuracy):} حذف لایه \lr{Pooling} و استفاده از گام حرکت
		(\lr{Stride}) باعث افت دقت نشد، بلکه در تمام حالت‌ها دقت به مقدار کمی
		افزایش یافت. بهترین عملکرد مربوط به مدل پنجم با دقت حدود 88 درصد است. استفاده از پدینگ \lr{same} باعث می‌شود اطلاعات لبه‌های تصویر در فرآیند کانولوشن
		از بین نرود و شبکه ویژگی‌های بهتری استخراج کند.
		
		\item \textbf{بررسی زمان آموزش:} مدل پایه که دارای \lr{Max Pooling} بود، سریع‌ترین
		زمان اجرا (حدود 139 ثانیه) را داشت. در لایه \lr{Pooling} هیچ وزن قابل یادگیری
		وجود ندارد، بنابراین محاسبات سریع‌تر انجام می‌شود.
		
		\item \textbf{تاثیر Stride نامتقارن:} در مدل‌های دوم و سوم که از استراید
		نامتقارن (1,2) و (2,1) استفاده شد، زمان آموزش به شدت افزایش یافت (به ترتیب
		حدود 313 و 267 ثانیه). دلیل این موضوع این است که تصویر تنها در یک بُعد فشرده
		می‌شود و خروجی لایه‌های کانولوشن بزرگ‌تر باقی می‌ماند. این کار باعث می‌شود
		تعداد پارامترها در لایه \lr{Dense} نهایی به شدت بالا برود و پردازش سنگین‌تر
		شود.
		
		\item \textbf{جایگزین منطقی Pooling:} مدل چهارم با \lr{Stride (2,2)} از نظر زمان
		آموزش (حدود 156 ثانیه) نزدیک‌ترین حالت به مدل پایه است؛ زیرا این استراید دقیقاً
		مانند یک لایه \lr{Pooling} (2x2)، ابعاد تصویر را در هر دو محور نصف کرده و حجم
		محاسبات را تا حد زیادی کنترل می‌کند.
	\end{itemize}

	\section{بخش دوم: بررسی بر روی داده‌های ارقام فارسی}
	در این بخش، آزمایش‌های انجام شده در بخش اول، بر روی مجموعه داده ارقام دست‌نویس
	فارسی هدی (\lr{Hoda Dataset}) تکرار شد. برای خواندن فایل‌های باینری اختصاصی این
	دیتاسِت (\lr{.cdb})، از اسکریپت استاندارد \lr{HodaDatasetReader} استفاده گردید
	و تصاویر به ابعاد ثابت ۲۸ در ۲۸ پیکسل تغییر اندازه داده شدند.
	
	\subsection{کد پیاده‌سازی و بارگذاری داده‌ها}
	قطعه کد زیر، نحوه بارگذاری فایل‌های \lr{cdb} و پیش‌پردازش آن‌ها را نشان می‌دهد.
	معماری شبکه عصبی و تنظیمات لایه‌ها دقیقاً مشابه بخش اول است.
	
	\begin{latin}
		\begin{lstlisting}
from HodaDatasetReader import read_hoda_cdb
import cv2
import numpy as np
import os

def load_hoda_cdb(dataset_path='hoda_dataset/Digits', target_size=(28, 28)):
train_path = os.path.join(dataset_path, 'Train 60000.cdb')
test_path = os.path.join(dataset_path, 'Test 20000.cdb')

train_images, train_labels = read_hoda_cdb(train_path)
test_images, test_labels = read_hoda_cdb(test_path)

def process_images(raw_images):
processed = [cv2.resize(img, target_size, interpolation=cv2.INTER_AREA) for img in raw_images]
processed_array = np.array(processed, dtype='float32') / 255.0
return np.expand_dims(processed_array, -1)

x_train = process_images(train_images)
x_test = process_images(test_images)
y_train = np.array(train_labels, dtype='int')
y_test = np.array(test_labels, dtype='int')

return (x_train, y_train), (x_test, y_test)
		\end{lstlisting}
	\end{latin}

	\section{بخش سوم: ترکیب ارقام و حروف فارسی}
	
	در آخرین بخش از این تمرین، مدل‌ها بر روی یک مجموعه داده بسیار پیچیده‌تر شامل ترکیبی
	از ارقام و حروف الفبای فارسی آموزش داده شدند. اضافه شدن حروف الفبا، چالش‌های
	جدیدی برای شبکه به همراه دارد؛ از جمله افزایش تعداد کلاس‌های خروجی (به ۱۹۶ کلاس)
	و شباهت ساختاری بسیار زیاد بین برخی از حروف (مانند ب، پ، ت، ث). در معماری شبکه،
	تنها تغییری که اعمال شد، تطبیق لایه \lr{Dense} نهایی با تعداد کلاس‌های جدید (مجموع
	ارقام و حروف) بود تا مدل قادر به پیش‌بینی احتمالات خروجی به درستی باشد.

	\begin{itemize}

		بخش (داده‌های انگلیسی، ارقام فارسی و حروف فارسی) الگوی ثابتی را نشان
		می‌دهد. جایگزینی لایه \lr{Pooling} با لایه کانولوشن با \lr{Stride (2,2)} به
		همراه پدینگ \lr{same} (مدل پنجم)، بهترین استراتژی برای حفظ و حتی ارتقای جزئیِ
		دقت مدل است (رسیدن به دقت 79.05 درصد).
		
		\item \textbf{افزایش شدید زمان محاسبات:} استفاده از \lr{Stride}های نامتقارن
		(مدل‌های دوم و سوم) به دلیل عدم کاهش متناسبِ ابعاد در خروجی لایه‌های
		کانولوشن، باعث تحمیل بار محاسباتی شدید به لایه‌های تمام‌متصل (\lr{Dense}) شده
		و زمان آموزش را به بیش از دو برابر زمان مدل پایه (حدود ۲۴۵ ثانیه) افزایش داده
		است.
