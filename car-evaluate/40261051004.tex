	\section{مقدمه و منطق جمع‌آوری خودکار داده‌ها}
	با توجه به خواسته تمرین مبنی بر ایجاد یک مجموعه داده با حداقل ۱۰۰ رکورد از اطلاعات خودروهای فروشی (شامل بیش از ۱۰ ویژگی مختلف برای هر ماشین)، ساده‌ترین راه در نگاه اول کپی کردن دستی اطلاعات از سایت‌های نیازمندی مانند دیوار بود. اما انجام دستی این کار چند مشکل اساسی داشت:
	\begin{itemize}
		\item \textbf{احتمال بالای خطای انسانی:} وارد کردن دستی حدود ۱۵۰۰ فیلد اطلاعاتی مختلف، احتمال اشتباه تایپی (جابه‌جا وارد کردن قیمت، کارکرد یا سال تولید) را به شدت افزایش می‌دهد که این موضوع در مرحله آموزش مدل باعث ایجاد نویز و افت دقت می‌شود.
		\item \textbf{زمان‌بر بودن:} استخراج دستی اطلاعات و خواندن دقیق متن توضیحات برای پیدا کردن ویژگی‌هایی مثل «ایربگ» یا «وضعیت لاستیک» نیازمند صرف ساعت‌ها زمان است.
		\item \textbf{عدم مقیاس‌پذیری:} در صورتی که در طول انجام پروژه متوجه می‌شدیم به داده‌های بیشتری نیاز داریم (مثلاً ۲۰۰ یا ۳۰۰ رکورد برای بهبود مدل)، باید مجدداً فرآیند خسته‌کننده دستی را تکرار می‌کردیم.
	\end{itemize}
	به همین دلایل، تصمیم بر آن شد تا یک اسکریپت وب‌اسکریپینگ به زبان پایتون نوشته شود تا داده‌ها به صورت کاملاً خودکار، ساختاریافته و با بالاترین دقت ممکن استخراج شوند. در نهایت تعداد ۱۵۰ رکورد از خودروی «تارا» از پلتفرم دیوار جمع‌آوری شد.
	
	\section{نحوه کارکرد اسکریپت استخراج داده (Scraping)}
	سایت‌های آگهی مانند دیوار برای جلوگیری از استخراج داده توسط ربات‌ها، از لایه‌های امنیتی مختلفی (مانند تغییر مداوم ساختار \lr{HTML} و سیستم‌های شناسایی ربات) استفاده می‌کنند. برای دور زدن این محدودیت‌ها و استخراج تمیز اطلاعات، رویکرد زیر در کد پیاده‌سازی شد:
	
	\begin{enumerate}
		\item \textbf{شبیه‌سازی مرورگر واقعی:} به جای استفاده از درخواست‌های ساده، از کتابخانه‌هایی استفاده شد که اثرانگشت \lr{(Fingerprint)} مرورگر کروم را شبیه‌سازی می‌کنند تا سیستم امنیتی سایت متوجه خودکار بودن فرآیند نشود.
		\item \textbf{استفاده از \lr{API} داخلی به جای \lr{HTML}:} به جای جستجو در تگ‌های تو در توی وب‌سایت که مدام در حال تغییر هستند، اسکریپت مستقیماً با \lr{API} بک‌اند سایت ارتباط برقرار می‌کند. با این کار، اطلاعات هر ماشین به صورت یک بسته داده‌ی خام و تمیز (\lr{JSON}) دریافت می‌شود.
		\item \textbf{پردازش متن توضیحات فروشنده:} ویژگی‌هایی مانند «وضعیت لاستیک‌ها» یا «وجود ایربگ» معمولاً در جدول مشخصات سایت دیوار وجود ندارند و فروشنده آن‌ها را در متن توضیحات می‌نویسد. در اسکریپت توابعی نوشته شد که متن توضیحات را اسکن کرده و این ویژگی‌ها را با استفاده از کلمات کلیدی استخراج کند.
	\end{enumerate}
	
	\section{ساختار داده‌های ذخیره شده (\lr{Dataset})}
	پس از پردازش اطلاعاتِ دریافت شده، داده‌ها به منظور استفاده در مدل‌های یادگیری ماشین، در یک فایل \lr{CSV} با نام \lr{divar\_tara\_ml\_dataset.csv} ذخیره شدند. فرمت \lr{CSV} بهترین استاندارد برای این کار است زیرا به سادگی توسط کتابخانه‌ی \lr{Pandas} در پایتون بارگذاری می‌شود.
	
	هر سطر از این فایل نمایانگر یک خودروی مجزا است و ستون‌های آن دقیقاً مطابق با نیازمندی‌های تمرین نگاشت شده‌اند:
	\begin{itemize}
		\item \textbf{مشخصات عددی:} قیمت، کارکرد (کیلومتر)، سال تولید و سن خودرو.
		\item \textbf{وضعیت سلامت:} بدنه، وضعیت موتور، وضعیت شاسی‌ها و گیربکس.
		\item \textbf{مشخصات جانبی:} رنگ، نوع سوخت، مهلت بیمه، وضعیت لاستیک و ایربگ.
	\end{itemize}
	در این مجموعه داده تلاش شده است تا داده‌های ناقص حذف شوند و متن‌های فارسی (مانند اعداد فارسی) برای جلوگیری از خطاهای پردازشی در آینده، پاک‌سازی و به اعداد استاندارد انگلیسی تبدیل شوند.

\section{ایجاد مجموعه داده و پیش‌پردازش}
در این تمرین، هدف بررسی قیمت خودروی «تارا» و پیش‌بینی آن بر اساس ویژگی‌های مختلف است. برای جمع‌آوری داده‌ها، به جای استخراج دستی که فرآیندی بسیار زمان‌بر و مستعد خطای انسانی (مانند اشتباه در وارد کردن اعداد طولانی قیمت) است، از یک اسکریپت خودکار برای استخراج داده‌ها (\lr{Web Scraper}) استفاده شد. 

این روش به ما اجازه داد تا در مدت زمانی کوتاه، اطلاعات دقیق ۱۵۰ خودرو را مستقیماً از پلتفرم دیوار دریافت کنیم. داده‌های استخراج شده برای دسترسی آسان در مراحل بعدی و قابلیت بارگذاری در کتابخانه‌های پردازشی، در قالب یک فایل \lr{CSV} ذخیره شدند. این فرمت به دلیل ساختار جدولی، بهترین گزینه برای نگهداری ویژگی‌های عددی و متنی به صورت همزمان است.

\section{مهندسی ویژگی‌ها و رگرسیون خطی}
پس از آماده‌سازی فایل داده‌ها، مراحل زیر جهت بهبود کیفیت یادگیری انجام شد:

\begin{itemize}
	\item \textbf{پاک‌سازی داده‌ها:} ستون‌هایی که تأثیری در قیمت نداشتند (مانند لینک آگهی و توکن) حذف شدند. همچنین با توجه به وجود ستون «سن خودرو»، ستون «سال تولید» برای جلوگیری از تکرار داده‌های مشابه حذف گردید.
	\item \textbf{حذف نویز:} در داده‌های وب، همواره مقادیر پرت وجود دارند. با اعمال فرمول $Q1 - 1.5 \times IQR$ و $Q3 + 1.5 \times IQR$، اسکریپت موفق شد ۹ رکورد نویز (شامل قیمت‌های بسیار پایین یا غیرمنطقی) را شناسایی و از چرخه آموزش حذف کند.
\end{itemize}

\subsection{نتایج مدل رگرسیون خطی}
مدل رگرسیون خطی به عنوان ساده‌ترین مدل برای شروع کار انتخاب شد. پس از آموزش مدل روی ۸۰ درصد داده‌ها، نتایج روی بخش تست به شرح زیر است:

\begin{itemize}
	\item \textbf{خطای میانگین مربعات (\lr{MSE}):} $7.88 \times 10^{16}$
	\item \textbf{ریشه خطای میانگین مربعات (\lr{RMSE}):} حدود ۲۸۰,۷۵۱,۲۵۹ تومان
\end{itemize}

این میزان خطا نشان می‌دهد که مدل خطی برای پیش‌بینی دقیق قیمت در بازار پرنوسان خودرو کافی نیست و لزوم استفاده از مدل‌های پیچیده‌تر مانند شبکه‌های عصبی در بخش‌های بعدی بررسی خواهد شد.
	
\section{بخش دوم: پیاده‌سازی شبکه عصبی با کتابخانه \lr{Numpy}}
در این بخش، مطابق با محدودیت‌های ذکر شده در صورت تمرین، یک شبکه عصبی پرسپترون چندلایه (\lr{MLP}) از پایه و تنها با استفاده از عملیات ماتریسی در کتابخانه \lr{Numpy} طراحی و پیاده‌سازی شد. در این مرحله استفاده از کتابخانه‌های سطح بالا مانند \lr{Keras} مجاز نبود.

\subsection{ساختار و ریاضیات شبکه}
برای جلوگیری از مشکل اشباع یا انفجار گرادیان در توابع فعال‌ساز (به دلیل بزرگ بودن ارقام ارقام مربوط به قیمت و کارکرد خودرو)، در ابتدا تمامی داده‌ها با استفاده از روش \lr{Min-Max Scaling} به بازه صفر تا یک نرمال شدند. ساختار شبکه شامل مراحل زیر است:
\begin{itemize}
	\item \textbf{انتشار رو به جلو (\lr{Forward Pass}):} ضرب ماتریسی ویژگی‌ها در وزن‌های هر لایه و عبور آن‌ها از تابع فعال‌ساز غیرخطی سیگموید (\lr{Sigmoid}).
	\item \textbf{محاسبه خطا:} محاسبه اختلاف میان قیمت پیش‌بینی شده و قیمت واقعی در فضای نرمال‌شده.
	\item \textbf{پس‌انتشار خطا (\lr{Backpropagation}):} محاسبه مشتقات جزئی خطا نسبت به وزن‌های هر لایه (با استفاده از قاعده زنجیره‌ای مشتق) و به‌روزرسانی تدریجی وزن‌ها با نرخ یادگیری (\lr{Learning Rate}) برابر با $0.05$.
\end{itemize}

\section{بخش سوم: پیاده‌سازی شبکه عصبی با \lr{Keras} و \lr{TensorFlow}}
در این مرحله، به منظور توسعه یک مدل مقیاس‌پذیر و استفاده از ابزارهای مدرن یادگیری عمیق، معماری شبکه عصبی با استفاده از کتابخانه \lr{Keras} پیاده‌سازی شد. استفاده از این فریم‌ورک امکان بهره‌گیری از بهینه‌سازهای پیشرفته و توابع فعال‌ساز نوین را فراهم می‌کند.

\subsection{معماری و پارامترهای مدل}
\begin{itemize}
	\item \textbf{نرمال‌سازی پیشرفته:} برخلاف روش دستی، در این بخش از کلاس \lr{MinMaxScaler} از کتابخانه \lr{scikit-learn} استفاده شد تا تمام ویژگی‌ها به صورت یکپارچه و امن به بازه $[0, 1]$ نگاشت شوند.
	\item \textbf{ساختار شبکه‌ی متوالی (\lr{Sequential}):}
	\begin{enumerate}
		\item \textbf{لایه ورودی و پنهان اول:} یک لایه \lr{Dense} شامل ۱۶ نود. به جای تابع کلاسیک سیگموید، از تابع فعال‌ساز \lr{ReLU} استفاده شد تا سرعت همگرایی افزایش یافته و مشکل محو شدگی گرادیان برطرف گردد.
		\item \textbf{لایه پنهان دوم:} شامل ۸ نود با تابع فعال‌ساز \lr{ReLU}.
		\item \textbf{لایه خروجی:} یک تک‌نود با تابع فعال‌ساز خطی (\lr{Linear}) که برای مسائل رگرسیون (پیش‌بینی مقادیر پیوسته مانند قیمت) مناسب‌ترین گزینه است.
	\end{enumerate}
	\item \textbf{الگوریتم بهینه‌ساز:} شبکه با استفاده از بهینه‌ساز پیشرفته‌ی \lr{Adam} کامپایل شد که نرخ یادگیری تطبیقی (\lr{Adaptive Learning Rate}) را برای هر پارامتر لحاظ می‌کند و عملکرد بسیار سریع‌تر و بهتری نسبت به گرادیان کاهشی استاندارد دارد.
\end{itemize}

\subsection{ارزیابی، چالش تنظیم \lr{Epoch} و پدیده بیش‌برازش}
در اجرای اولیه، مدل طی ۱۵۰ دوره (\lr{Epoch}) با اندازه دسته (\lr{Batch Size}) برابر با ۴ آموزش دید. خطای محاسبه شده بر روی داده‌های تست برای این حالت به شرح زیر بود:
\begin{itemize}
	\item \textbf{معیار خطای میانگین مربعات (\lr{MSE}):} $9.16 \times 10^{16}$
	\item \textbf{معیار خطای مجذور میانگین مربعات (\lr{RMSE}):} حدوداً ۳۰۲,۷۲۶,۳۵۵ تومان
\end{itemize}

با توجه به اینکه خطای این مدل از مدل دستیِ \lr{Numpy} (که ۱۵,۰۰۰ دوره آموزش دیده بود و خطای ۲۶۲ میلیونی داشت) بیشتر بود، در یک آزمایش جدید تصمیم گرفته شد تا تعداد دوره‌های آموزش (\lr{Epoch}) در کراس به ۸۰۰ افزایش یابد تا شبکه فرصت بیشتری برای یادگیری داشته باشد. 

\textbf{بروز بیش‌برازش (\lr{Overfitting}):}
پس از اجرای مدل با ۸۰۰ دوره، مشاهده شد که خطای آموزش (\lr{Train Loss}) به شدت کاهش یافت (به حدود $0.012$ رسید)، اما خطای تست به جای کاهش، \textbf{افزایش} یافت و \lr{RMSE} به رقم ۳۴۰ میلیون تومان رسید!

\textbf{تحلیل این رفتار:}
دلیل این اتفاق ترکیبِ «استفاده از بهینه‌ساز سریع \lr{Adam}» و «کوچک بودن مجموعه داده (حدود ۱۴۰ رکورد)» است. در مدل \lr{Numpy} به دلیل استفاده از مشتق‌گیری ساده و نرخ یادگیری ثابت، شبکه بسیار کند یاد می‌گرفت و به ۱۵,۰۰۰ دوره نیاز داشت. اما در \lr{Keras}، الگوریتم \lr{Adam} شبکه را به سرعت بهینه‌سازی می‌کند؛ بنابراین اعمال ۸۰۰ دوره آموزش برای این حجم کم از داده، باعث شد شبکه به جای کشف الگوهای کلی بازار، قیمت تک‌تک ماشین‌های آموزشی را «حفظ کند» (\lr{Memorization}). در نتیجه توانایی تعمیم خود را روی داده‌های تست از دست داد.

برای غلبه بر این مشکل و بهره‌گیری از حداکثر ظرفیت شبکه‌های عمیق بدون افتادن در دام بیش‌برازش، در بخش نهایی از تکنیک‌های منظم‌سازی (\lr{Regularization}) در یک معماری \lr{DNN} استفاده خواهیم کرد.

\section{بخش چهارم: شبکه عصبی عمیق (\lr{DNN}) و مقابله با بیش‌برازش}
برای حل چالش بیش‌برازش (\lr{Overfitting}) که در بخش قبل مشاهده شد و به منظور ارتقای عملکرد مدل روی داده‌های محدود، یک شبکه عصبی عمیق (\lr{Deep Neural Network}) با معماری پیشرفته‌تر پیاده‌سازی گردید.

\subsection{طراحی معماری \lr{DNN} و تکنیک‌های تنظیم (\lr{Regularization})}
معماری این شبکه شامل سه لایه مخفی به ترتیب با ۳۲، ۱۶ و ۸ نود است. برای جلوگیری از حفظ کردن داده‌ها، دو تکنیک اساسی در معماری شبکه لحاظ شد:

\begin{enumerate}
	\item \textbf{لایه دراپ‌اوت (\lr{Dropout}):} پس از لایه‌های مخفی اول و دوم، یک لایه \lr{Dropout} با نرخ $0.2$ اضافه شد. این لایه در هر مرحله از آموزش، به صورت تصادفی ۲۰ درصد از نرون‌ها را غیرفعال می‌کند تا شبکه به مسیرهای خاص وابسته نشده و قدرت تعمیم‌پذیری (\lr{Generalization}) آن افزایش یابد.
	\item \textbf{توقف زودهنگام (\lr{Early Stopping}):} به جای تعیین یک عدد ثابت و بزرگ برای \lr{Epoch} که منجر به بیش‌برازش می‌شود، از یک \lr{Callback} استفاده شد. با اختصاص ۲۰ درصد از داده‌های آموزش به عنوان داده‌های ارزیابی (\lr{Validation})، شبکه به صورت زنده رفتار خود را مانیتور می‌کند.
\end{enumerate}

\subsection{نتایج نهایی مدل عمیق}
با اجرای مدل با پیکربندی جدید، مکانیزم \lr{Early Stopping} متوجه شد که پس از حدود ۹۴ دوره (\lr{Epoch})، مدل به بهترین نقطه همگرایی خود رسیده است. بنابراین آموزش را به صورت خودکار متوقف کرد و وزن‌های بهترین دوره را بازگرداند. ارزیابی این شبکه بر روی داده‌های تست نهایی به شرح زیر است:

\begin{itemize}
	\item \textbf{معیار خطای میانگین مربعات (\lr{MSE}):} $6.82 \times 10^{16}$
	\item \textbf{ریشه خطای میانگین مربعات (\lr{RMSE}):} حدوداً ۲۶۱,۳۳۹,۴۳۳ تومان
\end{itemize}
